<h1>BC Consulting - Submit Your Business Info</h1>
<form id="businessForm" enctype="multipart/form-data">
  <label>Business Name</label>
  <input type="text" name="name" required>
  <label>Financials</label>
  <textarea name="financials"></textarea>
  <label>Other Info</label>
  <textarea name="otherInfo"></textarea>
  <label>Upload File</label>
  <input type="file" name="file">
  <button type="submit">Submit</button>
</form>

<script>
const form = document.getElementById('businessForm');
form.addEventListener('submit', async e => {
  e.preventDefault();
  const data = new FormData(form);
  const res = await fetch('/submit', { method:'POST', body:data });
  const result = await res.json();
  alert(result.message);
  form.reset();
});
</script>
